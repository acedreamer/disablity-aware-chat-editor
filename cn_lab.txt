1. TCP Socket Programming

TCP Server

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<arpa/inet.h>

int main()
{
    int s_sock,c_sock,len;
    struct sockaddr_in server,client;
    char buf[100];

    s_sock=socket(AF_INET,SOCK_STREAM,0);

    server.sin_family=AF_INET;
    server.sin_port=htons(3003);
    server.sin_addr.s_addr=INADDR_ANY;

    bind(s_sock,(struct sockaddr*)&server,sizeof(server));
    listen(s_sock,5);

    len=sizeof(client);
    c_sock=accept(s_sock,(struct sockaddr*)&client,&len);

    recv(c_sock,buf,sizeof(buf),0);
    printf("Message from client: %s\n",buf);

    printf("Enter reply: ");
    scanf("%s",buf);

    send(c_sock,buf,sizeof(buf),0);

    close(c_sock);
    close(s_sock);
}

2. UDP Socket Programming

UDP Server

#include<stdio.h>
#include<string.h>
#include<arpa/inet.h>
#include<sys/socket.h>

int main()
{
    int sock;
    struct sockaddr_in server,client;
    char buffer[100];
    int len;

    sock=socket(AF_INET,SOCK_DGRAM,0);

    server.sin_family=AF_INET;
    server.sin_port=htons(3004);
    server.sin_addr.s_addr=INADDR_ANY;

    bind(sock,(struct sockaddr*)&server,sizeof(server));

    len=sizeof(client);

    recvfrom(sock,buffer,100,0,(struct sockaddr*)&client,&len);
    printf("Message: %s\n",buffer);

    printf("Enter reply: ");
    scanf("%s",buffer);

    sendto(sock,buffer,100,0,(struct sockaddr*)&client,len);
}

3. Stop and Wait ARQ

Receiver (Server)

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<arpa/inet.h>

int main()
{
    int s_sock,c_sock,len;
    struct sockaddr_in server,client;
    char buf[100];
    int expected=0;

    s_sock=socket(AF_INET,SOCK_STREAM,0);

    server.sin_family=AF_INET;
    server.sin_port=htons(9009);
    server.sin_addr.s_addr=INADDR_ANY;

    bind(s_sock,(struct sockaddr*)&server,sizeof(server));
    listen(s_sock,5);

    len=sizeof(client);
    c_sock=accept(s_sock,(struct sockaddr*)&client,&len);

    while(1)
    {
        recv(c_sock,buf,sizeof(buf),0);
        int seq=buf[0]-'0';

        if(seq==expected)
        {
            printf("Received frame %d\n",seq);
            expected=1-expected;
        }
        else
            printf("Duplicate frame %d\n",seq);

        buf[0]=seq+'0';
        send(c_sock,buf,sizeof(buf),0);
    }
}

        send(sock,buf,sizeof(buf),0);

        recv(sock,buf,sizeof(buf),0);
        printf("ACK %c received\n",buf[0]);
    }
}

4. Go Back N ARQ

Receiver (Server)

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<arpa/inet.h>

int main()
{
    int s_sock,c_sock,len;
    struct sockaddr_in server,client;
    char buf[100];
    int expected=0;

    s_sock=socket(AF_INET,SOCK_STREAM,0);

    server.sin_family=AF_INET;
    server.sin_port=htons(9010);
    server.sin_addr.s_addr=INADDR_ANY;

    bind(s_sock,(struct sockaddr*)&server,sizeof(server));
    listen(s_sock,5);

    len=sizeof(client);
    c_sock=accept(s_sock,(struct sockaddr*)&client,&len);

    while(1)
    {
        recv(c_sock,buf,sizeof(buf),0);
        int seq=buf[0]-'0';

        if(seq==expected)
        {
            printf("Received frame %d\n",seq);
            expected++;
        }

        buf[0]=(expected-1)+'0';
        send(c_sock,buf,sizeof(buf),0);
    }
}


Sender (Client)

#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<unistd.h>
#include<arpa/inet.h>

#define WINDOW 4

int main()
{
    int sock,i=0,base=0;
    struct sockaddr_in server;
    char buf[100];

    sock=socket(AF_INET,SOCK_STREAM,0);

    server.sin_family=AF_INET;
    server.sin_port=htons(9010);
    server.sin_addr.s_addr=inet_addr("127.0.0.1");

    connect(sock,(struct sockaddr*)&server,sizeof(server));

    while(i<10)
    {
        if(i<base+WINDOW)
        {
            buf[0]=i+'0';
            printf("Sending frame %d\n",i);
            send(sock,buf,sizeof(buf),0);
            i++;
        }

        recv(sock,buf,sizeof(buf),0);
        int ack=buf[0]-'0';
        printf("ACK %d received\n",ack);
        base=ack+1;
    }
}

5. Distance Vector Routing Algorithm

#include<stdio.h>

int main()
{
    int n,i,j,k;
    int cost[10][10],dist[10],visited[10],min,u;

    printf("Enter number of nodes: ");
    scanf("%d",&n);

    printf("Enter cost matrix:\n");

    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            scanf("%d",&cost[i][j]);

    for(i=0;i<n;i++)
    {
        dist[i]=cost[0][i];
        visited[i]=0;
    }

    visited[0]=1;

    for(i=1;i<n;i++)
    {
        min=999;

        for(j=0;j<n;j++)
        {
            if(dist[j]<min && !visited[j])
            {
                min=dist[j];
                u=j;
            }
        }

        visited[u]=1;

        for(k=0;k<n;k++)
        {
            if(dist[k] > dist[u] + cost[u][k])
                dist[k] = dist[u] + cost[u][k];
        }
    }

    printf("Shortest distances:\n");

    for(i=0;i<n;i++)
        printf("%d\n",dist[i]);
}
